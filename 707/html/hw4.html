function xdot = zerodynamics(t, x)
    x1 = x(1);
    x2 = x(2);
    x3 = x(3);
    f = [x1*x2+x3; -2*x2; sin(x1)+2*x1*x2];
    g = [0; x1; 1];
    G = x1^2+1;
    dhdx = [1 0 0];
    xdot = (eye(3)-g*(1/G)*dhdx)*f;
end

function xdot = prop_to_ref(t, x, r)
    x1 = x(1);
    x2 = x(2);
    x3 = x(3);
    rdd = 0;
    e = r-x1;
    k = 2;
    f = [x1*x2+x3; -2*x2; sin(x1)+2*x1*x2];
    g = [0; x1; 1];
    G = x1^2+1;
    dhdx = [1 0 0];
    xdot = (eye(3)-g*(1/G)*dhdx)*f+g*(1/G)*(rdd + k*e);
end
